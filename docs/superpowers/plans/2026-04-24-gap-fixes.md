### Task 1: Fix file size check — read after size guard

**Files:**
- Modify: `backend/app/routers/ingest.py:240,309`

Both `ingest_official_pdf` and `ingest_unofficial_file` call `await file.read()` before the `MAX_FILE_SIZE` check, loading potentially 50 MB into RAM before rejecting it. Fix by checking `Content-Length` first and falling back to post-read check.

- [ ] **Step 1: Open `backend/app/routers/ingest.py` and find the two `await file.read()` call sites.**

They are at line ~240 (official PDF) and ~309 (unofficial file). Both follow this pattern:
```python

    raise HTTPException(...)
```

- [ ] **Step 2: Replace both with a size-safe read helper.**

Add this helper just above `_asset_type_from_mime` (around line 218):

```python
async def _safe_read(file: UploadFile, max_bytes: int) -> bytes:
    """Read upload rejecting early via Content-Length when available."""
    cl = file.headers.get("content-length")
    if cl and int(cl) > max_bytes:
        raise HTTPException(status_code=413, detail="File too large (max 50MB)")
    content = await file.read()
    if len(content) > max_bytes:
        raise HTTPException(status_code=413, detail="File too large (max 50MB)")
    return content
```

### Task 7: Fix overlap detection N+1 queries

**Files:**
- Modify: `backend/app/pipeline/overlap.py`

Currently `detect_overlaps` fetches all official questions in one query, then calls `db.get(QuestionAnnotation, oq.latest_annotation_id)` in a per-question loop — O(N) extra round trips. Fix with a single JOIN query.

- [ ] **Step 1: Rewrite `detect_overlaps` to use an outer join.**

Replace the current function body with:

```python
async def detect_overlaps(
    question_id: uuid.UUID,
    annotation_jsonb: dict,
    passage_text: Optional[str],
    question_text: str,
    db: AsyncSession,
    threshold: float = 0.4,
) -> List[Dict]:
    """Compare a question against all active official questions to detect overlaps.

    Uses a single JOIN query to load annotations alongside questions — avoids N+1.
    Returns: [{"official_question_id": UUID, "relation_type": str, "strength": float, "detection_method": str}]
    """
    from sqlalchemy.orm import outerjoin

    stmt = (
        select(Question, QuestionAnnotation)
        .outerjoin(
            QuestionAnnotation,
            Question.latest_annotation_id == QuestionAnnotation.id,
        )
        .where(
            Question.content_origin == "official",
            Question.practice_status == "active",
        )
    )
    rows = (await db.execute(stmt)).all()

    overlaps = []
    new_passage_words = _word_set(passage_text)
    new_question_words = _word_set(question_text)
    new_focus = annotation_jsonb.get("grammar_focus_key")

    for oq, official_annotation_row in rows:
        if oq.id == question_id:
            continue

        official_annotation = official_annotation_row.annotation_jsonb if official_annotation_row else None

        strength = 0.0
        detection_signals = []

        # Compare passage text
        if new_passage_words and oq.current_passage_text:
            official_passage_words = _word_set(oq.current_passage_text)
            passage_sim = _jaccard_similarity(new_passage_words, official_passage_words)
            if passage_sim > 0.3:
                strength = max(strength, passage_sim)
                detection_signals.append(f"passage_similarity={passage_sim:.2f}")

        # Compare question text
        if new_question_words:
            official_question_words = _word_set(oq.current_question_text)
            question_sim = _jaccard_similarity(new_question_words, official_question_words)
            if question_sim > 0.3:
                strength = max(strength, question_sim)
                detection_signals.append(f"question_similarity={question_sim:.2f}")

        # Compare grammar focus
        if official_annotation and new_focus:
            official_focus = official_annotation.get("grammar_focus_key")
            if official_focus == new_focus:
                strength = max(strength, 0.3)
                detection_signals.append("grammar_focus_match")

        if strength >= threshold:
            relation_type = "overlaps_official" if strength > 0.6 else "near_duplicate"
            overlaps.append({
                "official_question_id": oq.id,
                "relation_type": relation_type,
                "strength": round(strength, 3),
                "detection_method": "; ".join(detection_signals) if detection_signals else "automated",
            })

    return overlaps
```

```python
import uuid
from datetime import datetime, timezone
from typing import Optional

from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.database import get_db
from app.auth import admin_required, student_required
from app.models.db import User
from app.models.payload import UserCreate, UserResponse

router = APIRouter(prefix="/users", tags=["users"])


@router.post("", response_model=UserResponse, status_code=201)
async def create_user(
    body: UserCreate,
    db: AsyncSession = Depends(get_db),
    _auth: str = Depends(admin_required),
):
    existing = await db.execute(select(User).where(User.username == body.username))
    if existing.scalars().first():
        raise HTTPException(status_code=409, detail="Username already exists")
    user = User(username=body.username, created_at=datetime.now(timezone.utc))
    db.add(user)
    await db.commit()
    await db.refresh(user)
    return user


@router.get("", response_model=list[UserResponse])
async def list_users(
    limit: int = Query(50, ge=1, le=200),
    offset: int = Query(0, ge=0),
    db: AsyncSession = Depends(get_db),
    _auth: str = Depends(admin_required),
):
    result = await db.execute(
        select(User).order_by(User.id).offset(offset).limit(limit)
    )
    return result.scalars().all()


@router.get("/{user_id}", response_model=UserResponse)
async def get_user(
    user_id: int,
    db: AsyncSession = Depends(get_db),
    _auth: str = Depends(admin_required),
):
    user = await db.get(User, user_id)
    if not user:
        raise HTTPException(status_code=404, detail="User not found")
    return user


@router.delete("/{user_id}", status_code=204)
async def delete_user(
    user_id: int,
    db: AsyncSession = Depends(get_db),
    _auth: str = Depends(admin_required),
):
    user = await db.get(User, user_id)
    if not user:
        raise HTTPException(status_code=404, detail="User not found")
    await db.delete(user)
    await db.commit()
```
